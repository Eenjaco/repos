Auto-generate a readable changes overview (script)

Put this in Projects/Convert to markdown/generate_changes_overview.sh and run to create a markdown summary of commits touching that folder.

--

#!/usr/bin/env bash
# Generate a markdown overview of commits that touched the current folder.
# Usage: ./generate_changes_overview.sh [since] [outfile]
# Example: ./generate_changes_overview.sh "2025-01-01" CHANGES.md

set -euo pipefail

SINCE="${1:-}"
OUTFILE="${2:-CHANGES_AUTOGEN.md}"
FOLDER="$(pwd)"

echo "# Changes overview for $(basename "$FOLDER")" > "$OUTFILE"
echo "" >> "$OUTFILE"

GITARGS=(--no-merges --date=short --pretty=format:"## %ad — %s%n*Author:* %an%n%b" --name-only)
if [[ -n "$SINCE" ]]; then
  GITARGS=(--since="$SINCE" "${GITARGS[@]}")
fi

# run git log limited to this folder
git --no-pager log "${GITARGS[@]}" -- . | awk '
  BEGIN { in_commit=0 }
  /^## / { print ""; print $0; in_commit=1; next }
  /^*Author:/ { print $0; next }
  /^[[:space:]]*$/ { next }
  /^[^[:space:]]/ { print "* Message body: " $0; next }
  /^[[:space:]]/ { print "* Files changed:"; next }
  { print "  - " $0 }
' >> "$OUTFILE"

echo "Wrote $OUTFILE"
